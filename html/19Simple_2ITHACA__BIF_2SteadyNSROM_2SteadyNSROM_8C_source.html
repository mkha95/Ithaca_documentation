76  {
77  std::cout << green << "|F(x)| = " << res.norm() << " - Minimun reached in " <<
78  hnls.iter << " iterations " << def << std::endl << std::endl;
79  }
80  else if (Pstream::master())
81  {
82  std::cout << red << "|F(x)| = " << res.norm() << " - Minimun reached in " <<
83  hnls.iter << " iterations " << def << std::endl << std::endl;
84  }

87  tmp_sol<< (count_online_solve + 1),y;

94 void SteadyNSROM::reconstruct(fileName folder,const int & counter)
95  {
96  Info<<counter<<endl;
97  if (counter==1)
98  {
99  mkDir(folder);
100  ITHACAutilities::createSymLink(folder);
101  }
102  Eigen::MatrixXd CoeffU= online_solution[counter-1].block(1, 0, Nphi_u, 1);
103  Eigen::MatrixXd CoeffP= online_solution[counter-1].bottomRows(Nphi_p);
104  volVectorField uRec("uRec", Umodes[0]);
105  volScalarField pRec("pRec", problem->Pmodes[0]);
106  problem->L_U_SUPmodes.reconstruct(uRec, CoeffU, "uRec");
107  problem->Pmodes.reconstruct(pRec, CoeffP, "pRec");
108  uRecFields.append(uRec);
109  pRecFields.append(pRec);
110  ITHACAstream::exportSolution(uRec, name(counter), folder);
