128 {
129  print_section_name("OFFLINE SOLVE");
130  List<scalar> mu_now(1);
131  // if the offline solution is already performed read the fields
132  if (T::offline)
133  {
134  ITHACAstream::read_fields(T::Ufield,"U", "./ITHACAoutput/Offline/");
135  ITHACAstream::read_fields(T::Pfield,"p", "./ITHACAoutput/Offline/");
136  }
137 
138  else
139  {
140  for (label i = 0; i < mu.cols(); i++)
141  {
142  mu_now[0] = mu(0, i);
143  T::change_viscosity(mu(0, i));
144  auto start_ROM_REC = std::chrono::high_resolution_clock::now();
145  T::truthSolve(mu_now);
146  auto finish_ROM_REC = std::chrono::high_resolution_clock::now();
147  std::chrono::duration<double> elapsed = finish_ROM_REC - start_ROM_REC;
148  computation_times(i)=elapsed.count();
149  sampled_field(i)=sampling();
150  }
151  total_computation_time=computation_times.sum();
152  ITHACAstream::exportVector(sampled_field,"sampled_field","eigen","./ITHACAoutput/Offline");
153  ITHACAstream::exportVector(computation_times,"computation_times_FOM","eigen","./ITHACAoutput/Offline");
154  }

168  T::NUmodes = para->ITHACAdict->lookupOrDefault<scalar>("NmodesUproj", 10);
169  T::NPmodes = para->ITHACAdict->lookupOrDefault<scalar>("NmodesPproj", 10);
170  T::NSUPmodes = para->ITHACAdict->lookupOrDefault<scalar>("NmodesSUPproj", 10);
171  online= para->ITHACAdict->lookupOrDefault<word>("online","no");
172  if (T::bcMethod == "lift")
173  {

175  computeLift(T::Ufield, T::liftfield, T::Uomfield);
176  // Perform POD on the velocity snapshots
177  ITHACAPOD::getModes(T::Uomfield, T::Umodes, T::_U().name(),
178  T::podex, 0, 0, T::NUmodesOut);
179  }
180 
181  else
182  {
183  ITHACAPOD::getModes(T::Ufield, T::Umodes, T::_U().name(),
184  T::podex, 0, 0, T::NUmodesOut);
185  }
186 
187  // Perform POD on velocity pressure and supremizers and store the first modes
188  ITHACAPOD::getModes(T::Pfield, T::Pmodes, T::_p().name(),
189  T::podex, 0, 0,
190  T::NPmodesOut);
191  ITHACAPOD::getModes(T::supfield, T::supmodes, T::_U().name(),
192  T::podex,
193  T::supex, 1, T::NSUPmodesOut);
194  T::projectSUP("./Matrices", T::NUmodes, T::NPmodes, T::NSUPmodes);
195 
196  read_matrix(cumUeig,"./ITHACAoutput/POD/CumEigenvalues_U");
197  read_matrix(cumPeig,"./ITHACAoutput/POD/CumEigenvalues_p");
198  read_matrix(cumSupeig,"./ITHACAoutput/POD/CumEigenvalues_Usup");
199  double cum_eig_tol=0.9999;
200  bool check=true;
201  for(sugUmodes=0;sugUmodes<cumUeig.rows() && check;sugUmodes++)
202  check=(cumUeig(sugUmodes,0)<cum_eig_tol);
203 
204  check=true;
205  for(sugPmodes=0;sugPmodes<cumPeig.rows() && check;sugPmodes++)
206  check=(cumPeig(sugPmodes,0)<cum_eig_tol);
207 
208  check=true;
209  for(sugSupmodes=0;sugSupmodes<cumSupeig.rows() && check;sugSupmodes++)
210  check=(cumSupeig(sugSupmodes,0)<cum_eig_tol);
211 }

238 // Method to compute the lifting function
239 template <typename T>
240 void Bifurcation<T>::lift_solve()
241 {
242  if (lift_keyword=="potential")
243  {
244  print_section_name("SOLVING LIFTING PROBLEM");
245  surfaceScalarField& phi =T::_phi();
246  volScalarField& p = T::_p0();
247  volVectorField& U = T::_U0();
248  IOMRFZoneList& MRF = T::_MRF();
249  Time& runTime = T::_runTime();
250  fvMesh& mesh = T::_mesh();
251  label BCind = T::inletIndex(0, 0);
252  volVectorField Ulift("Ulift",U);
253  instantList Times = runTime.times();
254  runTime.setTime(Times[1], 1);
255  pisoControl potentialFlow(mesh, "potentialFlow");
256  Vector<double> v0(0, 0, 0);
257 
258  for (label j = 0; j < U.boundaryField().size(); j++)
259  {
260  if (j!=BCind)
261  {
262  T::assignBC(Ulift, j, v0);
263  }
264  }
265  T::assignIF(Ulift, v0);
266  phi = linearInterpolate(Ulift) & mesh.Sf();
267 
268  Info << "Constructing velocity potential field Phi\n" << endl;
269 
270  volScalarField Phi
271  (
272  IOobject
273  (
274  "Phi",
275  runTime.timeName(),
276  mesh,
277  IOobject::READ_IF_PRESENT,
278  IOobject::NO_WRITE
279  ),
280  mesh,
281  dimensionedScalar("Phi", dimLength * dimVelocity, 0),
282  p.boundaryField().types()
283  );
284  label PhiRefCell = 0;
285  scalar PhiRefValue = 0;
286  setRefCell
287  (
288  Phi,
289  potentialFlow.dict(),
290  PhiRefCell,
291  PhiRefValue
292  );
293  mesh.setFluxRequired(Phi.name());
294  runTime.functionObjects().start();
295  MRF.makeRelative(phi);
296  adjustPhi(phi, Ulift, p);
297 
298  while (potentialFlow.correctNonOrthogonal())
299  {
300  fvScalarMatrix PhiEqn
301  (
302  fvm::laplacian(dimensionedScalar("1", dimless, 1), Phi)
303  ==
304  fvc::div(phi)
305  );
306  PhiEqn.setReference(PhiRefCell, PhiRefValue);
307  PhiEqn.solve();
308 
309  if (potentialFlow.finalNonOrthogonalIter())
310  {
311  phi -= PhiEqn.flux();
312  }
313  }
314 
315  MRF.makeAbsolute(phi);
316  Info << "Continuity error = "
317  << mag(fvc::div(phi))().weightedAverage(mesh.V()).value()
318  << endl;
319  Ulift = fvc::reconstruct(phi);
320  Ulift.correctBoundaryConditions();
321  Info << "Interpolated velocity error = "
322  << (sqrt(sum(sqr((fvc::interpolate(U) & mesh.Sf()) - phi)))
323  / sum(mesh.magSf())).value()
324  << endl;
325  T::liftfield.append(Ulift);
326  }
327  else if (lift_keyword=="snapshot")
328  {
329  bool check_lift=ITHACAutilities::check_folder("Lift");
330  if (check_lift)
331  {
332  Info<<"Lifting function already exist: reading from folder \"./Lift/\""<<endl;
333  ITHACAstream::read_fields(this->liftfield, T::_U(), "./Lift/");
334  }
335  else
336  {
337  //in questo caso non abbiamo la cartella lift quindi la devo creare e devo anche prendere
338  //uno snapshot da usare come lift function
339  Info<<"\"Lift\" folder not found, creating \"./Lift/\" from first snapshot"<<endl;
340  ITHACAutilities::createSymLink("./Lift");
341  volVectorField& Ulift(T::Ufield[0]);
342  T::liftfield.append(Ulift);
343  ITHACAstream::exportSolution(Ulift, "1", "./Lift/");
344  }
345  }
346 }
