38 SimpleSteadyNSROM::SimpleSteadyNSROM(SteadyNSSimple& Foamproblem, Eigen::MatrixXd vel)
39  :
40  //calling the constructor of the base class
41  reducedSimpleSteadyNS(Foamproblem, vel),
42 
43  // inizializer list using relevant data from the FOM problem
44  P(Foamproblem._p()),
45  U(Foamproblem._U()),
46  mesh(Foamproblem._mesh()),
47  phi(Foamproblem._phi()),
48  Folder("./ITHACAoutput/Online"),
49  NmodesUproj(Foamproblem.NUmodes),
50  NmodesPproj(Foamproblem.NPmodes),
51  NmodesSup(Foamproblem.NSUPmodes),
52  NmodesNut(Foamproblem.NNutModes)
53 {
54 
55  maxIterOn=2000;
56 
57  //appending the modes from the FOM problem
58  ULmodes.resize(0);
59 
60  for (int i = 0; i < problem->inletIndex.rows(); i++)
61  {
62  ULmodes.append(problem->liftfield[i]);
63  }
64 
65  for (int i = 0; i < NmodesUproj; i++)
66  {
67  ULmodes.append(problem->Umodes.toPtrList()[i]);
68  }
69 
70  for (int i = 0; i < NmodesSup; i++)
71  {
72  ULmodes.append(problem->supmodes.toPtrList()[i]);
73  }
74 
75  // assigning the number of modes to be extracted
76  UprojN = NmodesUproj;
77  PprojN = NmodesPproj;
78 
79  // reading values for the residual jump from the ITHACAdict
80  residualJumpLim =
81  problem->para->ITHACAdict->lookupOrDefault<float>("residualJumpLim", 1e-5);
82  normalizedResidualLim =
83  problem->para->ITHACAdict->lookupOrDefault<float>("normalizedResidualLim",
84  1e-5);
85  residual_jump=1 + residualJumpLim;
86 
87  // inizialization of expansion coefficients and setting up boundary conditions
88  a = Eigen::VectorXd::Zero(UprojN);
89  b = Eigen::VectorXd::Zero(PprojN);
90  a(0) = vel_now(0, 0);
91 }

94 // REDUCED SIMPLE ALGORITHM solve for current value of the viscosity
95 void SimpleSteadyNSROM::solveOnline(scalar mu_now)
96 {
97  counter++;
98 
99  //changing value of the FOM problem, need to be done because the equation are then projected into the reduced space
100  problem->change_viscosity(mu_now);
101  if (counter==1)
102  {
103  mkDir(Folder);
104  ITHACAutilities::createSymLink(Folder);
105  }
106 
107  Eigen::VectorXd uresidualOld = Eigen::VectorXd::Zero(UprojN);
108  Eigen::VectorXd presidualOld = Eigen::VectorXd::Zero(PprojN);
109  Eigen::VectorXd uresidual = Eigen::VectorXd::Zero(UprojN);
110  Eigen::VectorXd presidual = Eigen::VectorXd::Zero(PprojN);
111  scalar U_norm_res(1);
112  scalar P_norm_res(1);
113 
114  Time& runTime = problem->_runTime();
115  P.rename("p");
116  ULmodes.reconstruct(U, a, "U");
117  problem->Pmodes.reconstruct(P, b, "p");
118  phi = fvc::interpolate(U) & U.mesh().Sf();
119  int iter = 0;
120  simpleControl& simple = problem->_simple();
121 
122 
123 
124  // eddy modes handling for future expansion with turbulence flow problems
125  if (ITHACAutilities::isTurbulent())
126  {
127 
128  Eigen::MatrixXd nutCoeff;
129  nutCoeff.resize(NmodesNut, 1);
130 
131  for (int i = 0; i < NmodesNut; i++)
132  {
133  Eigen::MatrixXd muEval;
134  muEval.resize(1, 1);
135  muEval(0, 0) = mu_now;
136  nutCoeff(i, 0) = problem->rbfSplines[i]->eval(muEval);
137  }
138 
139  volScalarField& nut = const_cast<volScalarField&>
140  (problem->_mesh().lookupObject<volScalarField>("nut"));
141  problem->nutModes.reconstruct(nut, nutCoeff, "nut");
142  ITHACAstream::exportSolution(nut, name(counter), Folder);
143  }
144 
145  PtrList<volVectorField> gradModP;
146 
147  for (int i = 0; i < NmodesPproj; i++)
148  {
149  gradModP.append(fvc::grad(problem->Pmodes[i]));
150  }
151 
152  projGradModP = ULmodes.project(gradModP, NmodesUproj);
153 
154  //Simple loop
155  while ((residual_jump > residualJumpLim
156  || std::max(U_norm_res, P_norm_res) > normalizedResidualLim)
157  && iter < maxIterOn)
158  {
159  iter++;
160  std::cout << "Iteration " << iter << std::endl;
161 #if OFVER == 6
162  simple.loop(runTime);
163 #else
164  simple.loop();
165 #endif
166  volScalarField nueff = problem->turbulence->nuEff();
167  fvVectorMatrix UEqn
168  (
169  fvm::div(phi, U)
170  - fvm::laplacian(nueff, U)
171  - fvc::div(nueff * dev2(T(fvc::grad(U))))
172  );
173  UEqn.relax();
174  List<Eigen::MatrixXd> RedLinSysU = ULmodes.project(UEqn, UprojN);
175  RedLinSysU[1] = RedLinSysU[1] - projGradModP * b;
176  a = reducedProblem::solveLinearSys(RedLinSysU, a, uresidual, vel_now);
177  ULmodes.reconstruct(U, a, "U");
178  volScalarField rAU(1.0 / UEqn.A());
179  volVectorField HbyA(constrainHbyA(1.0 / UEqn.A() * UEqn.H(), U, P));
180  surfaceScalarField phiHbyA("phiHbyA", fvc::flux(HbyA));
181  adjustPhi(phiHbyA, U, P);
182  tmp<volScalarField> rAtU(rAU);
183 
184  if (simple.consistent())
185  {
186  rAtU = 1.0 / (1.0 / rAU - UEqn.H1());
187  phiHbyA +=
188  fvc::interpolate(rAtU() - rAU) * fvc::snGrad(P) * mesh.magSf();
189  HbyA -= (rAU - rAtU()) * fvc::grad(P);
190  }
191 
192  List<Eigen::MatrixXd> RedLinSysP;
193 
194  while (simple.correctNonOrthogonal())
195  {
196  fvScalarMatrix pEqn
197  (
198  fvm::laplacian(rAtU(), P) == fvc::div(phiHbyA)
199  );
200  RedLinSysP = problem->Pmodes.project(pEqn, PprojN);
201  b = reducedProblem::solveLinearSys(RedLinSysP, b, presidual);
202  problem->Pmodes.reconstruct(P, b, "p");
203 
204  if (simple.finalNonOrthogonalIter())
205  {
206  phi = phiHbyA - pEqn.flux();
207  }
208  }
209 
210  P.relax();
211  U = HbyA - rAtU() * fvc::grad(P);
212  U.correctBoundaryConditions();
213  uresidualOld = uresidualOld - uresidual;
214  presidualOld = presidualOld - presidual;
215  uresidualOld = uresidualOld.cwiseAbs();
216  presidualOld = presidualOld.cwiseAbs();
217  residual_jump = std::max(uresidualOld.sum(), presidualOld.sum());
218  uresidualOld = uresidual;
219  presidualOld = presidual;
220  uresidual = uresidual.cwiseAbs();
221  presidual = presidual.cwiseAbs();
222  U_norm_res = uresidual.sum() / (RedLinSysU[1].cwiseAbs()).sum();
223  P_norm_res = presidual.sum() / (RedLinSysP[1].cwiseAbs()).sum();
224 
225  if (problem->para->debug)
226  {
227  std::cout << "Residual jump = " << residual_jump << std::endl;
228  std::cout << "Normalized residual = " << std::max(U_norm_res,
229  P_norm_res) << std::endl;
230  }
231  }
232 
233  std::cout << "Solution " << counter << " converged in " << iter <<
234  " iterations." << std::endl;
235  std::cout << "Final normalized residual for velocity: " << U_norm_res <<
236  std::endl;
237  std::cout << "Final normalized residual for pressure: " << P_norm_res <<
238  std::endl;
239 
240  // reconstruction of the full order solution
241  ULmodes.reconstruct(U, a, "Uaux");
242  P.rename("Paux");
243  problem->Pmodes.reconstruct(P, b, "Paux");
244  uRecFields.append(U);
245  pRecFields.append(P);
246 
247  // exporting the solution into the Online folder
248  ITHACAstream::exportSolution(U, name(counter), Folder);
249  ITHACAstream::exportSolution(P, name(counter), Folder);
250  runTime.setTime(runTime.startTime(), 0);
251 }
